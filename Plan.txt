from pymongo import MongoClient
import base64

# Connect to MongoDB
client = MongoClient('mongodb://localhost:27017/')
db = client['GCR_database']
collection = db['ChordCollection']

# Read binary data from image file
with open('c_major_tablature_1.jpg', 'rb') as f:
    binary_data = base64.b64encode(f.read())

# Sample chord document with binary data
chord_document = {
    "chordName": "C Major",
    "tablaturePictures": [
        {
            "data": binary_data,
            "description": "First inversion tablature"
        }
    ],
    "tutorialVideoLink": "https://www.example.com/c_major_tutorial"
}

# Insert document into collection
result = collection.insert_one(chord_document)
print("Inserted document ID:", result.inserted_id)
